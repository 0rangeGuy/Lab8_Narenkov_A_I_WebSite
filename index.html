Class Example

class Car: def __init__(self, make, model, year): self.make = make self.model = model self.year = year def display_info(self): return f"{self.year} {self.make} {self.model}" my_car = Car("Toyota", "Corolla", 2022) print(my_car.display_info())

Function Example

def calculate_total_cost(price, quantity, tax_rate): subtotal = price * quantity tax_amount = subtotal * tax_rate total_cost = subtotal + tax_amount return total_cost total = calculate_total_cost(10, 5, 0.08) print("Общая стоимость покупки:", total)
